def tokenizer(text):
  text = text.lower()
  rx1 = r"(?i)(?:(?<=\d)(?=[a-z])|(?<=[a-z])(?=\d))"
  text = re.sub(rx1,' ', text)
  text = re.sub(r'[^a-z0-9]',' ', text)
  text = ' '.join(text.split())
  text = text.split()
  return text
