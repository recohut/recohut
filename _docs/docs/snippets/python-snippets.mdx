## Clean filenames in a folder

```python
import re
import os
import pathlib
import glob

nbs = glob.glob('/content/docs/docs/tutorials/*.md')

def process(path):
    x = str(pathlib.Path(path).stem)
    x = x.lower()
    x = re.sub(r'[^a-z]','-',x)
    x = re.sub(r'-+','-',x)
    x = x.strip('-')
    x = os.path.join(str(pathlib.Path(path).parent), x+'.mdx')
    x = re.sub('/[a-z]\-','/',x)
    os.rename(path, x)

_ = [process(x) for x in nbs]
```
